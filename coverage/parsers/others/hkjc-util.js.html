const TIMEOUT = 2000;
 
const hkjc_login = async (page, credentials) => {
  console.log('üîê Starting HKJC login process...');
  
  // Navigate to HKJC Asian Handicap page
  console.log('üåê Navigating to HKJC Asian Handicap page...');
  await page.goto('https://bet.hkjc.com/en/football/hdc');
  await page.waitForTimeout(TIMEOUT);
  
  // Use EXACT working login method
  console.log('üìù Using EXACT login method...');
  await page.getByPlaceholder("Login name / Betting Account").click();
  await page.locator("#login-account-input").fill(credentials.username);
  await page.waitForTimeout(TIMEOUT);
  await page.locator("#login-account-input").press("Tab");
  await page.locator("#login-password-input").fill(credentials.password);
  await page.waitForTimeout(TIMEOUT);
  await page.getByText("Login").first().click();
  console.log('‚è∞ Waiting longer for login to process (can be slow)...');
  await page.waitForTimeout(TIMEOUT * 3); // 6 seconds instead of 2
 
  // Check for security question (may or may not appear)
  console.log('‚ùì Checking for security question...');
  try {
    const question = await page.locator(".login-question").textContent({ timeout: 5000 });
    if (question) {
      console.log("Security question found:", question);
      const answer = credentials.answers[question.trim()];
      
      if (answer) {
        console.log("Answering security question...");
        await page.locator('#betslip-panel').getByRole('textbox').fill(answer);
        await page.waitForTimeout(TIMEOUT);
        await page.getByText("Confirm", { exact: true }).click();
        await page.waitForTimeout(TIMEOUT);
      } else {
        console.log('‚ö†Ô∏è No answer found for security question');
      }
    }
  } catch (error) {
    console.log('‚ÑπÔ∏è No security question appeared, checking if already logged in...');
  }
 
  // confirm popup if any
  await page.getByText("Proceed").click();
  await page.waitForTimeout(TIMEOUT);
 
  // Verify login success by checking for logout button (wait longer as login can be slow)
  console.log('üîç Waiting for logout button to confirm login success (may take time)...');
  try {
    await page.getByText("Logout").waitFor({ timeout: 30000 }); // Wait up to 30 seconds
    console.log('‚úÖ Login successful - logout button found');
  } catch (error) {
    console.log('‚ö†Ô∏è Logout button not found, checking login status...');
    // Check if login form is still present (indicates login failed)
    const loginForm = await page.locator('#login-account-input').isVisible().catch(() => false);
    if (loginForm) {
      throw new Error('Login failed - login form still present');
    } else {
      console.log('‚ö†Ô∏è Login status unclear but no login form visible, proceeding...');
    }
  }
};
 
const hkjc_logout = async (page) => {
    console.warn('Logging out');
    await page.waitForTimeout(TIMEOUT);
    await page.getByText("Logout").click();
    await page.waitForTimeout(TIMEOUT);
    await page.getByTestId("overlay").getByText("OK").click();
    await page.waitForTimeout(TIMEOUT);
    await page.getByText("Close").click();
    console.warn('Logged out');
};
 
const hkjc_bet_handicap = async (page, match, amount = 200) => {
    console.warn("Placing bet", match.id, match.decision, amount);
    const index = match.decision === "home" ? 0 : 1;
    await page.locator(`#HDC_${match.id} input[type=checkbox]`).nth(index).click();
    await page.waitForTimeout(TIMEOUT);
    await page.getByText("Add").first().click();
    await page.waitForTimeout(TIMEOUT);
    await page.getByRole("textbox").dblclick();
    await page.waitForTimeout(TIMEOUT);
    await page.getByRole("textbox").fill(String(amount));
 
    await page.waitForTimeout(TIMEOUT);
    await page.getByRole("button", { name: "Place Bet" }).click();
    await page.waitForTimeout(TIMEOUT);
    await page.getByRole("button", { name: "Confirm" }).click();
    await page.waitForTimeout(TIMEOUT);
 
    await page.getByText("Done");
    await page.getByRole("button", { name: "Done" }).click();
}

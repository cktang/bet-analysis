    if (this.fileWatcher) {

  private setupFileWatcher() {
    // Watch for betting decisions to execute
    this.fileWatcher = chokidar.watch('./data/v2/betting-decisions.json');
    this.fileWatcher.on('change', async () => {
      console.log('💰 Betting decisions detected, executing bets...');
      await this.processBettingDecisions();
    });
  }
 
  private async processBettingDecisions() {
    try {
      const decisions = await this.dataFileService.readFile('betting-decisions.json');
      
      if (decisions.length === 0) return;
      
      const executionResults = [];
      
      for (const decision of decisions) {
        const result = await this.executeBet(decision);
        executionResults.push(result);
      }
      
      // Write execution results to file
      await this.dataFileService.writeFile('bet-execution-results.json', executionResults);
      
      // Archive processed decisions instead of clearing
      await this.archiveBettingDecisions(decisions);
      
      // Clear processed decisions
      await this.dataFileService.writeFile('betting-decisions.json', []);
      
      console.log(`💰 Executed ${executionResults.length} betting decisions`);
    } catch (error) {
      console.error('❌ Error processing betting decisions:', error);
    }
  }
 
  private async archiveBettingDecisions(decisions: any[]) {
    try {
      // Create archive directory if it doesn't exist
      const archiveDir = path.join(process.cwd(), 'data', 'v2', 'archive');
      if (!fs.existsSync(archiveDir)) {
        fs.mkdirSync(archiveDir, { recursive: true });
      }
      
      // Create timestamp for unique filename
      const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
      const archiveFilename = `bet-decisions-${timestamp}.json`;
      const archivePath = path.join(archiveDir, archiveFilename);
      
      // Write decisions to archive file
      fs.writeFileSync(archivePath, JSON.stringify(decisions, null, 2));
      
      console.log(`📁 Archived ${decisions.length} betting decisions to ${archiveFilename}`);
      
      // Also append to a master archive file
      const masterArchivePath = path.join(archiveDir, 'bet-decisions-master.json');
      let masterArchive = [];
      
      if (fs.existsSync(masterArchivePath)) {
        try {
          masterArchive = JSON.parse(fs.readFileSync(masterArchivePath, 'utf8'));
        } catch (error) {
          console.error('❌ Error reading master archive, creating new one:', error);
          masterArchive = [];
        }
      }
      
      // Add current decisions with metadata
      const archiveEntry = {
        timestamp: new Date().toISOString(),
        filename: archiveFilename,
        count: decisions.length,
        decisions: decisions
      };
      
      masterArchive.push(archiveEntry);
      
      // Keep only last 100 entries in master archive
      if (masterArchive.length > 100) {
        masterArchive = masterArchive.slice(-100);
      }
      
      fs.writeFileSync(masterArchivePath, JSON.stringify(masterArchive, null, 2));
      console.log(`📁 Updated master archive with ${decisions.length} new decisions`);
      
    } catch (error) {
      console.error('❌ Error archiving betting decisions:', error);
    }
  }
 
  async executeBet(signal: any): Promise<any> {
    if (this.isExecuting) {
      throw new Error('Another bet execution is in progress');
    }
    
    this.isExecuting = true;
    console.log(`💰 Executing bet: ${signal.homeTeam} vs ${signal.awayTeam} (${signal.betSide})`);
    
    try {
      const liveBettingEnabled = this.configService.get('ENABLE_LIVE_BETTING', false);
      const paperTradingEnabled = this.configService.get('ENABLE_PAPER_TRADING', true);
      
      if (liveBettingEnabled) {
        return await this.executeLiveBet(signal);
      } else if (paperTradingEnabled) {
        return await this.executePaperBet(signal);
      } else {
        throw new Error('Both live betting and paper trading are disabled');
      }
    } finally {
      this.isExecuting = false;
    }
  }

  private async initializeBrowser() {
    // Don't reinitialize if browser already exists and is connected
    if (this.browser && this.browser.pages) {
      try {
        const pages = this.browser.pages();
        if (pages.length > 0) {
          this.page = pages[0];
          console.log('♻️ Reusing existing persistent browser session');
          return;
        }
      } catch (error) {
        console.log('🔄 Browser session invalid, creating new one...');
        this.browser = null;
      }
    }
 
    // Create browser context with enhanced persistent storage to remember cookies/session
    const userDataDir = './data/v2/browser-session';
    
    // Ensure user data directory exists
    const fs = require('fs');
    if (!fs.existsSync(userDataDir)) {
      fs.mkdirSync(userDataDir, { recursive: true });
    }
    
    console.log('🚀 Creating persistent browser context for cookie persistence...');
    this.browser = await chromium.launchPersistentContext(userDataDir, {
      headless: this.configService.get('HEADLESS_BROWSER', true),
      args: [
        '--no-sandbox', 
        '--disable-setuid-sandbox',
        '--disable-blink-features=AutomationControlled',
        '--disable-web-security',
        '--disable-features=VizDisplayCompositor'
      ],
      viewport: { width: 1920, height: 1080 },
      // Enhanced cookie and session persistence
      ignoreHTTPSErrors: true,
      acceptDownloads: false,
      bypassCSP: true
    });
    
    // Use the default page from persistent context
    const pages = this.browser.pages();
    this.page = pages.length > 0 ? pages[0] : await this.browser.newPage();
    
    await this.page.setExtraHTTPHeaders({
      'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
    });
    
    console.log('✅ Persistent browser context created with cookie persistence');
  }
